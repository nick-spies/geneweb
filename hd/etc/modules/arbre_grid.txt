<script>
  document.addEventListener('DOMContentLoaded', function() {
    var dataEl = document.getElementById('tree-data');
    if (!dataEl) return;

    var text = dataEl.textContent;
    var data = { rows: [] };
    var currentRow = null;

    // Split on ~ROW~ and ~CELL~ markers
    var tokens = text.split('~ROW~');
    for (var i = 1; i < tokens.length; i++) {
      var rowChunk = tokens[i];
      var cellTokens = rowChunk.split('~CELL~');
      var isFirst = cellTokens[0].trim() === '1';
      currentRow = { isFirst: isFirst, cells: [] };

      for (var j = 1; j < cellTokens.length; j++) {
        var parts = cellTokens[j].split('~');
        var cell = {
          colspan: parseInt(parts[0]) || 1,
          isEmpty: parts[1] === '1',
          isTop: parts[2] === '1',
          isLeft: parts[3] === '1',
          isRight: parts[4] === '1'
        };

        if (!cell.isEmpty && parts[5]) {
          cell.person = {
            firstName: parts[5] || '',
            surname: parts[6] || '',
            publicName: parts[7] || '',
            access: parts[8] || '',
            sex: parseInt(parts[9]) || 0,
            hasImage: parts[10] === '1',
            imageUrl: parts[11] || '',
            hasSosa: parts[12] === '1',
            sosa: parts[13] || '',
            birthYear: parts[14] || '',
            deathYear: parts[15] || '',
            hasParents: parts[17] === '1'
          };
          if (cell.isRight) {
            cell.family = { marriageYear: parts[16] || '' };
          }
        }

        currentRow.cells.push(cell);
      }

      data.rows.push(currentRow);
    }

    try {
      var container = document.getElementById('tree-grid-container');
      var renderer = new TreeRenderer(container, data, {
        prefix: '%prefix;',
        wizard: %if;wizard;true%else;false%end;,
        generations: %e_v;,
        selfAccess: '%prefix;pz=%if;(evar.pz!="")%evar.pz;%else;%evar.p;%end;&nz=%if;(evar.nz!="")%evar.nz;%else;%evar.n;%end;&p=%if;(evar.pz!="")%evar.pz;%else;%evar.p;%end;&n=%if;(evar.nz!="")%evar.nz;%else;%evar.n;%end;&ocz=%evar.ocz;',
        selfName: '%if;(evar.pz!="")%evar.pz;%else;%evar.p;%end; %if;(evar.nz!="")%evar.nz;%else;%evar.n;%end;',
        currentName: '%self;',
        isRerooted: %if;(evar.pz!="" and evar.pz!=evar.p)true%else;false%end;
      });
      renderer.render();
    } catch(e) {
      console.error('Tree render error:', e);
      document.getElementById('tree-grid-container').innerHTML =
        '<div class="alert alert-danger">Error rendering tree: ' + e.message + '</div>';
    }
  });
</script>
