      if (!(e.metaKey || e.ctrlKey)) return;
      if (e.target.closest('a') || e.target.closest('button')) return;

    var maxR = 0, maxB = 0;

      var b = boxes[i];
      var bt = parseInt(b.style.top), bl = parseInt(b.style.left);
      var bw = parseInt(b.style.width), bh = parseInt(b.style.height);
      if (isNaN(bt) || isNaN(bl)) continue;

      if (bt + bh > maxB) maxB = bt + bh;

    // Click to select nearest person. Tests both box dots (isoScreenPts)
    // and bar segments (isoBarSegs) so clicking a tall bar selects the

    // Point-in-convex-quad: cross-product sign test
    function ptInQuad(px, py, pts) {
      var pos = 0, neg = 0;
      for (var i = 0; i < 4; i++) {
        var j = i < 3 ? i + 1 : 0;
        var a = pts[i], b = pts[j];
        var cross = (b.x - a.x) * (py - a.y) - (b.y - a.y) * (px - a.x);
        if (cross > 0) pos++; else if (cross < 0) neg++;
      }
      return pos === 0 || neg === 0;
    }
    function isoFindNearest(cx, cy) {
      var best = null, bestD = 20;  // max 20px hit distance
      // Test projected box quads first (ground-plane parallelograms)
      for (var qi = 0; qi < isoBoxQuads.length; qi++) {
        var q = isoBoxQuads[qi];
        if (ptInQuad(cx, cy, q.pts)) {
          // Inside a box — use distance to center as tie-breaker
          var sp = isoScreenPts[q.sosa];
          if (sp) {
            var qd = Math.sqrt((sp.x - cx) * (sp.x - cx) + (sp.y - cy) * (sp.y - cy));
            if (!best || qd < bestD) { bestD = qd; best = q.sosa; }
          } else {
            best = q.sosa; bestD = 0;
          }
        }
      }
      if (best) return best;
      // Fall back to dot proximity
      for (var sid in isoScreenPts) {
        var sp = isoScreenPts[sid];
        var d = Math.sqrt((sp.x - cx) * (sp.x - cx) + (sp.y - cy) * (sp.y - cy));
        if (d < bestD) { bestD = d; best = sid; }
      }
      // Also test distance from click to each bar segment.
      for (var bi = 0; bi < isoBarSegs.length; bi++) {
        var seg = isoBarSegs[bi];
        var ax = seg.foot.x, ay = seg.foot.y;
        var bx = seg.top.x, by = seg.top.y;
        var dx = bx - ax, dy = by - ay;
        var len2 = dx * dx + dy * dy;
        var t = len2 > 0 ? ((cx - ax) * dx + (cy - ay) * dy) / len2 : 0;
        if (t < 0) t = 0; else if (t > 1) t = 1;
        var px = ax + dx * t, py = ay + dy * t;
        var segD = Math.sqrt((px - cx) * (px - cx) + (py - cy) * (py - cy));
        // Hit tolerance: bar half-width + 4px slack
        var tol = (seg.halfW || 3) + 4;
        if (segD < tol && segD < bestD) { bestD = segD; best = String(seg.sosa); }
      }
      return best;
    }

    // Shift+arrow: find nearest dot in the chart direction projected through iso rotation

    function isoFindNearestDir(fromSosa, dir) {
      var from = isoScreenPts[fromSosa];
      var dv = isoChartDirs[dir];
      if (!from || !dv) return null;
      // Normalize the chart direction vector
      var mag = Math.sqrt(dv.x * dv.x + dv.y * dv.y);
      if (mag === 0) return null;
      var ux = dv.x / mag, uy = dv.y / mag;
      var best = null, bestScore = Infinity;
      for (var sid in isoScreenPts) {
        if (sid === fromSosa) continue;
        var sp = isoScreenPts[sid];
        var dx = sp.x - from.x, dy = sp.y - from.y;
        // Project onto chart direction: must be positive (in the direction we're heading)
        var along = dx * ux + dy * uy;
        if (along < 2) continue;
        // Perpendicular distance
        var perp = Math.abs(dx * (-uy) + dy * ux);
        var score = along + perp * 2;
        if (score < bestScore) { bestScore = score; best = sid; }
      }
      return best;
    }

        // Plain arrow: fixed genealogical navigation
        // ← father, → mother, ↓ child (same as main chart)
        // ↓ at local sosa 1 pops back to the previous iso chart (if any),
        // selecting the dSosa we just left — which is now an ancestor
        // somewhere on that chart's plane. Continuing ↓ keeps stepping
        // toward the original oSosa; stops when we reach Sosa 1 of the
        // base (first-loaded) chart.
        if (e.key === 'ArrowDown' && s === 1 && isoGenStack.length > 0) {
          var popped = isoGenStack[isoGenStack.length - 1];
          var parentUrl = popped && popped.from;
          if (parentUrl) {
            // Pop the stack entry, stash the popped sosa as pending selection
            // for the parent chart, then forward-navigate (no browser back).
            isoGenStack.pop();
            isoSaveStack();
            try {
              sessionStorage.setItem('hiIsoPendingSel', String((popped && popped.sosa) || 1));
              sessionStorage.setItem('hiIsoAngle', String(isoAngle));
              sessionStorage.setItem('hiIsoTilt', String(isoTilt));
            } catch(_) {}
            window.location.href = parentUrl;
          }
          return;
        }
        if (e.key === 'ArrowLeft') next = s * 2;
        else if (e.key === 'ArrowRight') next = s * 2 + 1;
        else if (e.key === 'ArrowDown' && s > 1) next = Math.floor(s / 2);
        else return;
        next = isoScreenPts[String(next)] ? next : null;
      }

      if (next) {
        e.preventDefault();
        isoPushHistory(isoSelectedSosa);
        isoSelectedSosa = String(next);
        drawIso();
        isoEnsureSelectedVisible();
        isoUpdateInfo();
      }
    }, true);

    function drawIso() {
      isoCtx.clearRect(0, 0, isoW, isoH);
      // Keep zoom label in sync
      var _zl = document.getElementById('hi-zoom-level');
      if (_zl) _zl.textContent = Math.round(isoZoom * 100) + '%%';

      // ── Collect data ──
      var topAncs = tree.querySelectorAll('a[data-depth]');
      var maxDepth = 0;
      var baseGen = 0;  // generation of the leaves of the rendered tree (wrt oSosa 1)
      var barPts = [];
      for (var i = 0; i < topAncs.length; i++) {
        var d = parseInt(topAncs[i].dataset.depth) || 0;
        if (d > maxDepth) maxDepth = d;
        var box = document.getElementById('s' + topAncs[i].dataset.sosa);
        if (!box) continue;
        var bx = parseInt(box.style.left), by = parseInt(box.style.top);
        var bw2 = parseInt(box.style.width), bh2 = parseInt(box.style.height);
        if (isNaN(bx) || isNaN(by)) continue;
        var sosa = parseInt(topAncs[i].dataset.sosa) || 0;
        // bars are emitted for leaf-level persons; their 0-indexed gen is
        // floor(log2(sosa)). Take the max across bars to get the tree's leaf gen.
        if (sosa > 0) {
          var bg = Math.floor(Math.log(sosa) / Math.log(2));
          if (bg > baseGen) baseGen = bg;
        }
        barPts.push({ x: bx + bw2 / 2, y: by + bh2 / 2, depth: d, sosa: sosa });
      }
      var hasExtraDepth = maxDepth > baseGen;
      if (maxDepth === 0) maxDepth = 1;
      var minDepth = baseGen;
      var depthMap = {};
      for (var i = 0; i < barPts.length; i++) depthMap[barPts[i].sosa] = barPts[i].depth;
      // Find deepest local gen of any person on the tree plane
      var treeMaxGen = 0;
      var allAncs = tree.querySelectorAll('a[data-sosa]');
      for (var ai = 0; ai < allAncs.length; ai++) {
        var as = parseInt(allAncs[ai].dataset.sosa) || 0;
        if (as > 0) {
          var ag = Math.floor(Math.log(as) / Math.log(2));
          if (ag > treeMaxGen) treeMaxGen = ag;
        }
      }
      // Highest gen available from this iso view (absolute from oSosa):
      //   - plane persons: treeMaxGen + isoGenOffset
      //   - bar persons: maxDepth (data-depth is already absolute)
      var maxAvailGen = Math.max(maxDepth, treeMaxGen + isoGenOffset);
      // depthSpan = distance from base to top tick. Capped at isoSelGens
      // but limited to the highest available gen.
      var depthSpan = Math.max(1, Math.min(isoSelGens, maxAvailGen - isoGenOffset));

      // ── Rotatable isometric projection ──
      // 3D world:  X = tree horizontal,  Z = tree vertical (into scene),  Y = bar height (up)
      // Step 1: rotate (X, Z) by isoAngle around vertical axis
      // Step 2: tilt by isoTilt (~35.264°) to get isometric foreshortening
      // Step 3: orthographic projection to screen
      var cosA = Math.cos(isoAngle), sinA = Math.sin(isoAngle);
      var cosT = Math.cos(isoTilt), sinT = Math.sin(isoTilt);
      var PAD = 12;
      // Scale post height proportionally so tick spacing per gen stays
      // constant across charts. Full height (0.28) at isoSelGens gens.
      var barMaxH = 0.084 * (depthSpan / isoSelGens);

      // Normalize tree coords to [0..1] range for uniform scaling
      var normX = treeW > 0 ? 1.0 / treeW : 1;
      var normZ = treeH > 0 ? 1.0 / treeH : 1;

      // Raw projection: rotate around Y, then tilt around X, then drop Z
      function rawProj(x, z, y) {
        var nx = x * normX - 0.5, nz = z * normZ - 0.5;  // center at origin
        // Rotate around Y axis by isoAngle
        var rx = nx * cosA - nz * sinA;
        var rz = nx * sinA + nz * cosA;
        // Tilt around X axis by isoTilt; negate X for correct chirality when viewed from above
        var screenX = -rx;
        var screenY = -(rz * sinT + (y || 0)) ;
        return { x: screenX, y: screenY };
      }

      // Find bounding box to fit into canvas
      var corners = [rawProj(0,0,0), rawProj(treeW,0,0), rawProj(0,treeH,0), rawProj(treeW,treeH,0),
                     rawProj(0,0,barMaxH), rawProj(treeW,0,barMaxH), rawProj(0,treeH,barMaxH), rawProj(treeW,treeH,barMaxH)];
      var minSX = corners[0].x, maxSX = corners[0].x, minSY = corners[0].y, maxSY = corners[0].y;
      for (var ci = 1; ci < corners.length; ci++) {
        if (corners[ci].x < minSX) minSX = corners[ci].x;
        if (corners[ci].x > maxSX) maxSX = corners[ci].x;
        if (corners[ci].y < minSY) minSY = corners[ci].y;
        if (corners[ci].y > maxSY) maxSY = corners[ci].y;
      }
      var projW = maxSX - minSX, projH = maxSY - minSY;
      // Use max possible projW (diagonal) so scale doesn't change with rotation
      var maxProjW = Math.SQRT2;  // normalized coords span [-0.5,0.5] in both axes
      var fitScale = ((isoW - PAD * 2) / maxProjW) * isoZoom;

      // Anchor on Sosa 1: find its tree position, pin it at a fixed screen point
      var sosa1Box = document.getElementById('s1');
      var s1tx = treeW / 2, s1tz = treeH / 2;  // fallback: center
      if (sosa1Box) {
        s1tx = parseInt(sosa1Box.style.left) + parseInt(sosa1Box.style.width) / 2;
        s1tz = parseInt(sosa1Box.style.top) + parseInt(sosa1Box.style.height) / 2;
      }
      var s1proj = rawProj(s1tx, s1tz, 0);
      // Pin Sosa 1 at screen center horizontally, 50% from top vertically
      var pinX = isoW / 2, pinY = isoH * 0.50;
      var offX = pinX - s1proj.x * fitScale + isoPanX;
      var offY = pinY - s1proj.y * fitScale + isoPanY;

      // Project tree position (tx, tz) with bar height fraction h (0..1) to screen
      function toScreen(tx, tz, h) {
        var rp = rawProj(tx, tz, (h || 0) * barMaxH);
        return { x: rp.x * fitScale + offX, y: rp.y * fitScale + offY };
      }

      // ── Compute chart direction vectors in screen space ──
      // Flat chart: Left=fathers(x=0), Right=mothers(x=treeW), Up=gen0(y=0), Down=genN(y=treeH)
      // Project unit chart vectors through iso rotation for Shift+arrow navigation
      var cMid = toScreen(treeW / 2, treeH / 2, 0);
      var cLeft = toScreen(0, treeH / 2, 0);           // father side
      var cRight = toScreen(treeW, treeH / 2, 0);      // mother side
      var cUp = toScreen(treeW / 2, 0, 0);             // earliest gen
      var cDown = toScreen(treeW / 2, treeH, 0);       // latest gen
      // Map arrow keys to iso chart orientation (matches flat chart directly)
      // Left → fathers(x=0), Right → mothers, Up → gen0(y=0), Down → genN(y=treeH)
      isoChartDirs = {
        ArrowLeft:  { x: cLeft.x - cMid.x, y: cLeft.y - cMid.y },
        ArrowRight: { x: cRight.x - cMid.x, y: cRight.y - cMid.y },
        ArrowUp:    { x: cUp.x - cMid.x, y: cUp.y - cMid.y },
        ArrowDown:  { x: cDown.x - cMid.x, y: cDown.y - cMid.y }
      };

      // ── Build box position lookup ──
      var allBoxes = tree.querySelectorAll('[id^="s"]');
      var boxPos = {};
      for (var i = 0; i < allBoxes.length; i++) {
        var b = allBoxes[i];
        var sid = b.id.substring(1);
        var bt2 = parseInt(b.style.top), bl2 = parseInt(b.style.left);
        var bw3 = parseInt(b.style.width), bh3 = parseInt(b.style.height);
        if (isNaN(bt2) || isNaN(bl2)) continue;
        boxPos[sid] = { x: bl2 + (bw3 || 0) / 2, y: bt2 + (bh3 || 0) / 2, w: bw3 || 0, h: bh3 || 0, bg: b.style.backgroundColor || '' };
      }

      // If only one person on the chart, center them on the iso plane
      var boxKeys = Object.keys(boxPos);
      if (boxKeys.length === 1) {
        var onlyKey = boxKeys[0];
        var ob = boxPos[onlyKey];
        boxPos[onlyKey] = { x: treeW / 2, y: treeH / 2, w: ob.w, h: ob.h, bg: ob.bg };
      }

      // ── Re-root: map each subtree member to the position its corresponding
      // display-sosa would occupy in the original chart. This keeps the same
      // visual scale as the original iso view — the re-rooted person R takes
      // the center slot (where oSosa 1 was), its parents take the slots of
      // sosa 2 and 3, etc. Original box dimensions preserved.
      if (isoRoot > 1 && boxPos[String(isoRoot)]) {
        var rootBits = Math.floor(Math.log(isoRoot) / Math.log(2));
        var remapped = {};
        for (var sid in boxPos) {
          var sv = parseInt(sid);
          if (!isoInSubtree(sv, isoRoot)) continue;
          var svBits = Math.floor(Math.log(sv) / Math.log(2));
          var d = svBits - rootBits;
          var twoD = Math.pow(2, d);
          // sv = isoRoot * 2^d + j, where 0 <= j < 2^d; dSosa m = 2^d + j
          var j = sv - isoRoot * twoD;
          var m = twoD + j;
          var pos = boxPos[String(m)];
          if (!pos) continue;
          remapped[sid] = { x: pos.x, y: pos.y, w: pos.w, h: pos.h, bg: boxPos[sid].bg };
        }
        // Scale up the re-rooted subtree so it fills the plane at a natural size.
        // The subtree occupies only the slot positions 1..2^(evarv-rootBits)-1,
        // which is a small portion of the original plane. Scale uniformly, but
        // cap to prevent huge boxes when subtree is very small.
        var rKeys = Object.keys(remapped);
        if (rKeys.length > 1) {
          var rMinX = Infinity, rMinY = Infinity, rMaxX = -Infinity, rMaxY = -Infinity;
          for (var ki = 0; ki < rKeys.length; ki++) {
            var rp = remapped[rKeys[ki]];
            if (rp.x - rp.w / 2 < rMinX) rMinX = rp.x - rp.w / 2;
            if (rp.x + rp.w / 2 > rMaxX) rMaxX = rp.x + rp.w / 2;
            if (rp.y - rp.h / 2 < rMinY) rMinY = rp.y - rp.h / 2;
            if (rp.y + rp.h / 2 > rMaxY) rMaxY = rp.y + rp.h / 2;
          }
          var rW2 = Math.max(1, rMaxX - rMinX);
          var rH2 = Math.max(1, rMaxY - rMinY);
          var rSc = Math.min(treeW / rW2, treeH / rH2, 4);
          var rOfX = (treeW - rW2 * rSc) / 2 - rMinX * rSc;
          var rOfY = (treeH - rH2 * rSc) / 2 - rMinY * rSc;
          for (var ki2 = 0; ki2 < rKeys.length; ki2++) {
            var rsp = remapped[rKeys[ki2]];
            remapped[rKeys[ki2]] = { x: rsp.x * rSc + rOfX, y: rsp.y * rSc + rOfY, w: rsp.w * rSc, h: rsp.h * rSc, bg: rsp.bg };
          }
        }
        boxPos = remapped;
        // Filter depthMap to subtree and recompute max. depth values stay
        // in absolute oSosa-1 gen coords. baseGen for the subtree = max
        // gen of leaf-level boxes that have bars in the subtree.
        var newDepthMap = {};
        var newMaxDepth = 0;
        var newBaseGen = 0;
        for (var dk in depthMap) {
          if (isoInSubtree(parseInt(dk), isoRoot)) {
            var dv = depthMap[dk];
            newDepthMap[dk] = dv;
            if (dv > newMaxDepth) newMaxDepth = dv;
            var bg2 = Math.floor(Math.log(parseInt(dk)) / Math.log(2));
            if (bg2 > newBaseGen) newBaseGen = bg2;
          }
        }
        depthMap = newDepthMap;
        if (newMaxDepth > 0) maxDepth = newMaxDepth;
        if (newBaseGen > 0) minDepth = newBaseGen;
        hasExtraDepth = maxDepth > minDepth;
        var maxAvailGen2 = Math.max(maxDepth, treeMaxGen + isoGenOffset);
        depthSpan = Math.max(1, Math.min(isoSelGens, maxAvailGen2 - isoGenOffset));
        barMaxH = 0.14 * (depthSpan / isoSelGens);
        // Also filter barPts to subtree members and remap their x,y to the
        // new box positions so bars sit on top of the re-rooted boxes.
        var newBarPts = [];
        for (var bi = 0; bi < barPts.length; bi++) {
          var bp = barPts[bi];
          if (!isoInSubtree(bp.sosa, isoRoot)) continue;
          var np = boxPos[String(bp.sosa)];
          if (!np) continue;
          newBarPts.push({ x: np.x, y: np.y, depth: bp.depth, sosa: bp.sosa });
        }
        barPts = newBarPts;
      }

      // ── Draw connecting lines (parent-child) on ground plane ──
      // Blue for father lines (even sosa), red for mother lines (odd sosa)
      // Line width = 60% of the narrow side of the (smaller) box, in screen pixels.
      var isoNarrowPx = {};  // sid -> projected narrow-side px
      for (var npid in boxPos) {
        var nbp = boxPos[npid];
        var nhw = nbp.w / 2, nhh = nbp.h / 2;
        var nc0 = toScreen(nbp.x - nhw, nbp.y - nhh, 0);
        var nc1 = toScreen(nbp.x + nhw, nbp.y - nhh, 0);
        var nc3 = toScreen(nbp.x - nhw, nbp.y + nhh, 0);
        var ndx1 = nc1.x - nc0.x, ndy1 = nc1.y - nc0.y;
        var ndx2 = nc3.x - nc0.x, ndy2 = nc3.y - nc0.y;
        var nl1 = Math.sqrt(ndx1 * ndx1 + ndy1 * ndy1);
        var nl2 = Math.sqrt(ndx2 * ndx2 + ndy2 * ndy2);
        isoNarrowPx[npid] = Math.min(nl1, nl2);
      }
      for (var sid in boxPos) {
        var s = parseInt(sid);
        if (s <= 1) continue;
        var parentId = String(Math.floor(s / 2));
        if (!boxPos[parentId]) continue;
        var cp = toScreen(boxPos[sid].x, boxPos[sid].y, 0);
        var pp = toScreen(boxPos[parentId].x, boxPos[parentId].y, 0);
        var lineW = Math.min(isoNarrowPx[sid] || 4, isoNarrowPx[parentId] || 4) * 0.4;
        if (lineW < 1) lineW = 1;
        isoCtx.lineWidth = lineW;
        isoCtx.strokeStyle = (s % 2 === 0) ? 'rgba(74,138,199,0.5)' : 'rgba(208,96,128,0.5)';
        isoCtx.beginPath();
        isoCtx.moveTo(cp.x, cp.y);
        isoCtx.lineTo(pp.x, pp.y);
        isoCtx.stroke();
      }

      // ── Draw all person boxes as projected parallelograms on ground plane ──
      isoScreenPts = {};
      isoBoxQuads = [];
      for (var sid in boxPos) {
        var bp = boxPos[sid];
        var s2 = parseInt(sid);
        // Gender: even sosa = male (square), odd = female (rounded). Root: check flat box color.
        var isMale;
        if (s2 > 1) {
          isMale = (s2 % 2 === 0);
        } else {
          var rootBox = document.getElementById('s' + sid);
          var rbg = (rootBox && rootBox.style.backgroundColor || '').toLowerCase();
          isMale = rbg.indexOf('208, 228, 245') >= 0 || rbg.indexOf('d0e4f5') >= 0;
        }
        var isSel = (sid === isoSelectedSosa);
        // Selected box: fully saturated color. Others: light shade.
        isoCtx.fillStyle = isSel
          ? (isMale ? '#3070a8' : '#b04870')
          : (isMale ? '#A8CDEB' : '#EBB2C0');
        var sp = toScreen(bp.x, bp.y, 0);
        isoScreenPts[sid] = { x: sp.x, y: sp.y };
        // Project box corners through isometric transform
        var hw = bp.w / 2, hh = bp.h / 2;
        var c0 = toScreen(bp.x - hw, bp.y - hh, 0);
        var c1 = toScreen(bp.x + hw, bp.y - hh, 0);
        var c2 = toScreen(bp.x + hw, bp.y + hh, 0);
        var c3 = toScreen(bp.x - hw, bp.y + hh, 0);
        isoBoxQuads.push({ sosa: sid, pts: [c0, c1, c2, c3] });
        isoCtx.beginPath();
        if (isMale) {
          isoCtx.moveTo(c0.x, c0.y);
          isoCtx.lineTo(c1.x, c1.y);
          isoCtx.lineTo(c2.x, c2.y);
          isoCtx.lineTo(c3.x, c3.y);
          isoCtx.closePath();
        } else {
          // Rounded corners for females: quadratic curve through each vertex.
          var e01x = c1.x - c0.x, e01y = c1.y - c0.y;
          var e12x = c2.x - c1.x, e12y = c2.y - c1.y;
          var e23x = c3.x - c2.x, e23y = c3.y - c2.y;
          var e30x = c0.x - c3.x, e30y = c0.y - c3.y;
          var L01 = Math.sqrt(e01x * e01x + e01y * e01y) || 1;
          var L12 = Math.sqrt(e12x * e12x + e12y * e12y) || 1;
          var L23 = Math.sqrt(e23x * e23x + e23y * e23y) || 1;
          var L30 = Math.sqrt(e30x * e30x + e30y * e30y) || 1;
          var r = Math.min(L01, L12, L23, L30) * 0.35;
          if (r > 10) r = 10;
          // Shrink endpoints of each edge inward by r
          var a0x = c0.x + e01x / L01 * r, a0y = c0.y + e01y / L01 * r;
          var b0x = c1.x - e01x / L01 * r, b0y = c1.y - e01y / L01 * r;
          var a1x = c1.x + e12x / L12 * r, a1y = c1.y + e12y / L12 * r;
          var b1x = c2.x - e12x / L12 * r, b1y = c2.y - e12y / L12 * r;
          var a2x = c2.x + e23x / L23 * r, a2y = c2.y + e23y / L23 * r;
          var b2x = c3.x - e23x / L23 * r, b2y = c3.y - e23y / L23 * r;
          var a3x = c3.x + e30x / L30 * r, a3y = c3.y + e30y / L30 * r;
          var b3x = c0.x - e30x / L30 * r, b3y = c0.y - e30y / L30 * r;
          isoCtx.moveTo(a0x, a0y);
          isoCtx.lineTo(b0x, b0y);
          isoCtx.quadraticCurveTo(c1.x, c1.y, a1x, a1y);
          isoCtx.lineTo(b1x, b1y);
          isoCtx.quadraticCurveTo(c2.x, c2.y, a2x, a2y);
          isoCtx.lineTo(b2x, b2y);
          isoCtx.quadraticCurveTo(c3.x, c3.y, a3x, a3y);
          isoCtx.lineTo(b3x, b3y);
          isoCtx.quadraticCurveTo(c0.x, c0.y, a0x, a0y);
          isoCtx.closePath();
        }
        isoCtx.fill();
        // Subtle border
        isoCtx.strokeStyle = 'rgba(0,0,0,0.15)';
        isoCtx.lineWidth = 0.5;
        isoCtx.stroke();
        // Sosa number inside box when it fits
        var edgeW = Math.sqrt((c1.x - c0.x) * (c1.x - c0.x) + (c1.y - c0.y) * (c1.y - c0.y));
        var edgeH = Math.sqrt((c3.x - c0.x) * (c3.x - c0.x) + (c3.y - c0.y) * (c3.y - c0.y));
        var sosaStr = sid;
        var charW = Math.min(edgeW, edgeH) * 0.4;
        if (charW >= 4 && sosaStr.length * charW * 0.6 < edgeW * 0.9) {
          isoCtx.save();
          var ax = c1.x - c0.x, ay = c1.y - c0.y;
          var cx = c3.x - c0.x, cy = c3.y - c0.y;
          var sc = charW;
          var ux = ax / edgeW, uy = ay / edgeW;
          var vx = cx / edgeH, vy = cy / edgeH;
          isoCtx.setTransform(ux * sc, uy * sc, vx * sc, vy * sc, sp.x, sp.y);
          isoCtx.fillStyle = isSel ? '#fff' : 'rgba(0,0,0,0.55)';
          isoCtx.font = 'bold 1px sans-serif';
          isoCtx.textAlign = 'center';
          isoCtx.textBaseline = 'middle';
          isoCtx.fillText(sosaStr, 0, 0);
          isoCtx.restore();
        }
      }

      // ── Highlight ancestor path on ground plane (before bars so bars draw on top) ──
      if (isoSelectedSosa && isoScreenPts[isoSelectedSosa]) {
        var path = [];
        var cs = parseInt(isoSelectedSosa);
        while (cs > 1) {
          var ps = Math.floor(cs / 2);
          var csp = isoScreenPts[String(cs)], psp = isoScreenPts[String(ps)];
          if (csp && psp) path.push({ cs: cs, from: csp, to: psp });
          cs = ps;
        }
        // Pass 1: white halo
        isoCtx.strokeStyle = '#fff';
        isoCtx.lineWidth = Math.max(3, 9 * Math.sqrt(isoZoom));
        for (var pi = 0; pi < path.length; pi++) {
          isoCtx.beginPath();
          isoCtx.moveTo(path[pi].from.x, path[pi].from.y);
          isoCtx.lineTo(path[pi].to.x, path[pi].to.y);
          isoCtx.stroke();
        }
        // Pass 2: gender-colored line
        isoCtx.lineWidth = Math.max(2, 6 * Math.sqrt(isoZoom));
        for (var pi = 0; pi < path.length; pi++) {
          isoCtx.strokeStyle = (path[pi].cs % 2 === 0) ? '#4a8ac7' : '#d06080';
          isoCtx.beginPath();
          isoCtx.moveTo(path[pi].from.x, path[pi].from.y);
          isoCtx.lineTo(path[pi].to.x, path[pi].to.y);
          isoCtx.stroke();
        }
      }

      // ── Edge labels (drawn on ground plane, so bars occlude them) ──
      var labelWorldSz = 0.035;
      function drawPlaneLabel(wx, wy, label, color, adx, ady, cdx, cdy) {
        var orig = toScreen(wx, wy, 0);
        var aEnd = toScreen(wx + adx * labelWorldSz, wy + ady * labelWorldSz, 0);
        var cEnd = toScreen(wx + cdx * labelWorldSz, wy + cdy * labelWorldSz, 0);
        var ax = aEnd.x - orig.x, ay = aEnd.y - orig.y;

  function toSup(n) {
    if (n < 2) return '';
    var s = '', ds = String(n);
    for (var i = 0; i < ds.length; i++) s += _sup[parseInt(ds[i])];

  function rleGender(syms) {
    var compact = '', cur = syms[0], cnt = 1;
    for (var i = 1; i < syms.length; i++) {
      if (syms[i] === cur) cnt++;
      else { compact += cur + toSup(cnt); cur = syms[i]; cnt = 1; }
    }
    return compact + cur + toSup(cnt);
  }

  // Build node position cache for spatial navigation
  var nodeCache = [];
  function buildCache() {

  // Get dot center position for a sosa number
  function dotPos(sosa) {
    var d = document.getElementById('s' + sosa);
    if (!d) return null;
    return { x: d.offsetLeft + d.offsetWidth / 2, y: d.offsetTop + d.offsetHeight / 2 };

  // Spatial nav follows connected nodes: father(s*2), mother(s*2+1), child(s/2)
  // Maps arrow key to whichever connected node lies in that direction
  function findConnected(s, cx, cy, dir) {
    var neighbors = [];
    var f = dotPos(s * 2);
    if (f) neighbors.push({ sosa: s * 2, dx: f.x - cx, dy: f.y - cy });
    var m = dotPos(s * 2 + 1);
    if (m) neighbors.push({ sosa: s * 2 + 1, dx: m.x - cx, dy: m.y - cy });
    if (s > 1) {
      var c = dotPos(Math.floor(s / 2));
      if (c) neighbors.push({ sosa: Math.floor(s / 2), dx: c.x - cx, dy: c.y - cy });
    }
    // Score each neighbor: how well does it match the arrow direction?
    var best = null, bestScore = -Infinity;
    for (var i = 0; i < neighbors.length; i++) {
      var n = neighbors[i];
      var score = 0;
      if (dir === 'ArrowUp') score = -n.dy;
      else if (dir === 'ArrowDown') score = n.dy;
      else if (dir === 'ArrowLeft') score = -n.dx;
      else if (dir === 'ArrowRight') score = n.dx;
      if (score > 2 && score > bestScore) { bestScore = score; best = n.sosa; }
    }
    return best;
  }

  // Find nearest node in a screen direction from any position (fallback when no connected node)
  function findNearest(cx, cy, dir, currentSosa) {
    var best = null, bestDist = Infinity;
    for (var i = 0; i < nodeCache.length; i++) {
      var n = nodeCache[i];
      if (n.el.dataset.sosa === String(currentSosa)) continue;
      var dx = n.x - cx, dy = n.y - cy;
      var ok = false;
      if (dir === 'ArrowUp' && dy < -2) ok = true;
      else if (dir === 'ArrowDown' && dy > 2) ok = true;
      else if (dir === 'ArrowLeft' && dx < -2) ok = true;
      else if (dir === 'ArrowRight' && dx > 2) ok = true;
      if (!ok) continue;
      var along, cross;
      if (dir === 'ArrowUp' || dir === 'ArrowDown') {
        along = Math.abs(dy); cross = Math.abs(dx);
      } else {
        along = Math.abs(dx); cross = Math.abs(dy);
      }
      var dist = along + cross * 5;
      if (dist < bestDist) { bestDist = dist; best = n.el; }
    }
    return best;
  }

  // On load: mark all implex nodes blue and add cross-references to tooltips
  function markImplexes() {
    var byHref = {};
    var els = tree.querySelectorAll('a[data-sosa]');
    for (var i = 0; i < els.length; i++) {
      var h = personKey(els[i]);
      if (!byHref[h]) byHref[h] = [];
      byHref[h].push({ sosa: els[i].dataset.sosa, el: els[i] });
    }
    for (var h in byHref) {
      var group = byHref[h];
      if (group.length > 1) {
        var sosas = group.map(function(g) { return g.sosa; });
        for (var j = 0; j < group.length; j++) {
          // fuzzy blue outline on implex nodes (preserves gender background)
          var dot = document.getElementById('s' + group[j].sosa);
          if (dot) {
            dot.style.boxShadow = '0 0 3px 2px rgba(30,144,255,0.7)';
          }
          // add cross-ref to title if missing
          var el = group[j].el;
          var others = sosas.filter(function(s) { return s !== group[j].sosa; });
          var t = el._hiTitle || el.getAttribute('title') || '';
          var ref = ' (=' + others.join('|') + ')';
          // replace existing partial cross-ref or add new one
          t = t.replace(/ \(=[0-9,|]+\)/, '');
          // match sosa number with or without comma formatting
          var sn = group[j].sosa;
          var snc = Number(sn).toLocaleString();
          t = t.replace('Sosa ' + snc, 'Sosa ' + snc + ref);
          if (t.indexOf(ref) < 0) t = t.replace('Sosa ' + sn, 'Sosa ' + sn + ref);
          el._hiTitle = t;
        }
      }
    }
  }

  // Ghost boxes: faint placeholders beyond chart edge for ancestors with parents
  function addGhostBoxes() {
    // For each last-gen ancestor with parents beyond the chart, place ghost boxes
    // where father (sosa*2) and mother (sosa*2+1) would appear.
    // H-tree geometry: next-gen offset = parent-to-child offset rotated 90° and halved.
    var ghostOutline = (new URLSearchParams(location.search).get('ghost') === 'outline');
    var all = tree.querySelectorAll('a[data-sosa]');
    var maxGen = 0;
    for (var i = 0; i < all.length; i++) {
      var g = Math.floor(Math.log2(parseInt(all[i].dataset.sosa)));
      if (g > maxGen) maxGen = g;
    }
    for (var i = 0; i < all.length; i++) {
      var s = parseInt(all[i].dataset.sosa);
      if (Math.floor(Math.log2(s)) !== maxGen) continue;
      if (!all[i].dataset.hasf && !all[i].dataset.hasm) continue;
      var box = document.getElementById('s' + s);
      if (!box) continue;
      var bx = box.offsetLeft, by = box.offsetTop;
      var bw = box.offsetWidth, bh = box.offsetHeight;
      var parentSosa = Math.floor(s / 2);
      var parentBox = document.getElementById('s' + parentSosa);
      if (parentBox) {
        var px = parentBox.offsetLeft, py = parentBox.offsetTop;
        var dx = bx - px, dy = by - py;
        var nx = -dy / 2, ny = dx / 2;
        var gx1 = bx + nx, gy1 = by + ny;
        var gx2 = bx - nx, gy2 = by - ny;
        var base = Math.max(bw, bh);
        var tri = Math.max(6, Math.round(base / 4));
        var hasF = all[i].dataset.hasf === '1';
        var hasM = all[i].dataset.hasm === '1';
        var colors = ['rgba(100,160,220,0.45)', 'rgba(220,100,140,0.45)'];
        var strokeColors = ['rgba(100,160,220,0.7)', 'rgba(220,100,140,0.7)'];
        for (var gi = 0; gi < 2; gi++) {
          if (gi === 0 && !hasF) continue;
          if (gi === 1 && !hasM) continue;
          var gx = gi === 0 ? gx1 : gx2;
          var gy = gi === 0 ? gy1 : gy2;
          var col = colors[gi];
          var strokeCol = strokeColors[gi];
          var offx = gx - bx, offy = gy - by;
          var tx, ty, tw, th, pts;
          if (Math.abs(offx) > Math.abs(offy)) {
            var triBase = Math.round(bh * 0.6);
            var pad = Math.round((bh - triBase) / 2);
            tw = tri; th = triBase;
            ty = by + pad;
            if (offx > 0) {
              tx = bx + bw;
              pts = '0,0 ' + tw + ',' + Math.round(th/2) + ' 0,' + th;
            } else {
              tx = bx - tri;
              pts = tw + ',0 0,' + Math.round(th/2) + ' ' + tw + ',' + th;
            }
          } else {
            var triBase = Math.round(bw * 0.6);
            var pad = Math.round((bw - triBase) / 2);
            tw = triBase; th = tri;
            tx = bx + pad;
            if (offy > 0) {
              ty = by + bh;
              pts = '0,0 ' + tw + ',0 ' + Math.round(tw/2) + ',' + th;
            } else {
              ty = by - tri;
              pts = Math.round(tw/2) + ',0 ' + tw + ',' + th + ' 0,' + th;
            }
          }
          var svgNS = 'http://www.w3.org/2000/svg';
          var svg = document.createElementNS(svgNS, 'svg');
          svg.setAttribute('width', tw);
          svg.setAttribute('height', th);
          svg.style.cssText = 'position:absolute;top:' + ty + 'px;left:' + tx + 'px;z-index:0;pointer-events:none;overflow:visible;';
          var poly = document.createElementNS(svgNS, 'polygon');
          poly.setAttribute('points', pts);
          if (ghostOutline) {
            poly.setAttribute('fill', 'none');
            poly.setAttribute('stroke', strokeCol);
            poly.setAttribute('stroke-width', '1.5');
          } else {
            poly.setAttribute('fill', col);
            poly.setAttribute('stroke', strokeCol);
            poly.setAttribute('stroke-width', '1');
          }
          svg.appendChild(poly);
          svg.classList.add('hi-ghost');
          tree.appendChild(svg);
        }
      }
    }
    // Expand tree container if ghosts overflow
    var treW = parseInt(tree.style.width) || tree.offsetWidth;
    var treH = parseInt(tree.style.height) || tree.offsetHeight;
    var ghosts = tree.querySelectorAll('.hi-ghost');
    for (var i = 0; i < ghosts.length; i++) {
      var g = ghosts[i];
      var r = g.offsetLeft + g.offsetWidth + 10;
      var b = g.offsetTop + g.offsetHeight + 10;
      if (r > treW) treW = r;
      if (b > treH) treH = b;
    }
    tree.style.width = treW + 'px';
    tree.style.height = treH + 'px';
  }

    if (!m) return;

    var els = tree.querySelectorAll('a[data-sosa]');
    for (var i = 0; i < els.length; i++) {
      var t = els[i]._hiTitle || els[i].getAttribute('title') || '';
      var gm = t.match(/^Gen (\d+)/);
      if (gm) { var g = parseInt(gm[1]); if (g > maxGen) maxGen = g; }
    }
    if (maxGen > 0 && maxGen < parseInt(m[2])) {
      h3.textContent = h3.textContent.replace(/\(([^)]*\s)\d+(\s*\))/, '($1' + maxGen + '$2');
    }
  }

          var targetUrl = boundary.href;
          if (/([?&])dsosa=\d+/.test(targetUrl)) {
            targetUrl = targetUrl.replace(/([?&])dsosa=\d+/, '$1dsosa=' + gsosa);
          } else {
            var sep = targetUrl.indexOf('?') >= 0 ? '&' : '?';
            targetUrl += sep + 'dsosa=' + gsosa;
          }
          sessionStorage.setItem('hiIsoPendingSearch', '' + n);
          // Push gen stack so ↓ can walk back toward Sosa 1
          if (window.hiIsoPushGenStack) {
            window.hiIsoPushGenStack(boundary.dataset.sosa, targetUrl);
          }
          try {
            sessionStorage.setItem('hiIsoAngle', String(window.hiIsoGetAngle ? window.hiIsoGetAngle() : 0));
            sessionStorage.setItem('hiIsoTilt', String(window.hiIsoGetTilt ? window.hiIsoGetTilt() : 0));
            sessionStorage.setItem('hiIsoPendingSel', '1');
          } catch(_) {}
          window.location.href = targetUrl;
        } else {
          // Can't find any ancestor on chart — person doesn't exist
          inp.style.background = '#fdd';
          setTimeout(function() { inp.style.background = ''; }, 800);
        }
      } else {
        // Flat chart: not on current display — increase generations and reload
        var absGen = Math.floor(Math.log2(n));
        var localGen = absGen + 1;
        var params = new URLSearchParams(location.search);
        var curGen = parseInt(params.get('v')) || 7;
        if (localGen > curGen) {
          sessionStorage.setItem('hi-active-sosa', '' + n);
          params.set('v', '' + localGen);
          window.location.search = params.toString();
        } else {
          // enough generations but person not found (doesn't exist in tree)
          inp.style.background = '#fdd';
          setTimeout(function() { inp.style.background = ''; }, 800);
        }
      }
    };
    var sosaInput = document.getElementById('hi-sosa-search');
    if (sosaInput) sosaInput.addEventListener('keydown', function(e) {
      if (e.key !== 'Enter') return;
      e.preventDefault();
      e.stopPropagation();
      window.hiSosaSearch();
    });
  })();

  // Hover: show tooltip (Cmd held = "Cmd+click: go to person")
  var hoveredA = null;
  var origTitle = '';
  var glowEls = [];
  function clearGlow() {
    for (var i = 0; i < glowEls.length; i++) glowEls[i].classList.remove('hi-glow');
    glowEls = [];
  }
  var scrollGuard = false, hoverTimer = null;
  tree.addEventListener('mouseover', function(e) {
    var a = e.target.closest('a[data-sosa]');
    if (!a || scrollGuard) return;
    if (hoveredA !== a) {
      hoveredA = a;
      origTitle = a._hiTitle || '';
      if (hoverTimer) clearTimeout(hoverTimer);
      hoverTimer = setTimeout(function() {
        if (hoveredA !== a) return;
        showInfo(a);
        // Show Sosa balloon if number is truncated in box
        var numSpan = a.querySelector('.hi-sosa-num');
        if (numSpan && numSpan.scrollWidth > numSpan.clientWidth) {
          hoverBalloon.textContent = numSpan.textContent.trim();
          var box = a.closest('div[id^="s"]');
          if (box) {
            var bx = box.offsetLeft, by = box.offsetTop, bw = box.offsetWidth;
            hoverBalloon.style.display = 'block';
            hoverBalloon.style.left = (bx + bw / 2 - hoverBalloon.offsetWidth / 2) + 'px';
            hoverBalloon.style.top = (by - hoverBalloon.offsetHeight - 6) + 'px';
          }
        } else {
          hoverBalloon.style.display = 'none';
        }
      }, 300);
    }
  });
  tree.addEventListener('mouseout', function(e) {
    var a = e.target.closest('a[data-sosa]');
    if (!a) return;
    var related = e.relatedTarget;
    if (related && a.contains(related)) return;
    if (hoveredA === a) {
      if (hoverTimer) { clearTimeout(hoverTimer); hoverTimer = null; }
      hoveredA = null; origTitle = '';
      hoverBalloon.style.display = 'none';
    }
  });

  // Click = select; Cmd+click = go to person page; Enter = re-root
  tree.addEventListener('click', function(e) {
    var a = e.target.closest('a[data-sosa]');
    if (!a) return;
    e.preventDefault();
    if (e.metaKey || e.ctrlKey) {
      var url = a.dataset.purl;
      if (url) window.location.href = url;
    } else {
      activate(a);
    }
  });

  // Track Cmd key state for tooltip hints (info bar always shows full info)
  var cmdHeld = false;
  document.addEventListener('keydown', function(e) {
    if (e.key === 'Meta' || e.key === 'Control') {
      cmdHeld = true;
      if (hoveredA && origTitle) {
        var name = origTitle.split('\n')[1] || '';
        hoveredA.setAttribute('title', HI_T.cmd_click + name.trim());
      }
    }
  });
  document.addEventListener('keyup', function(e) {
    if (e.key === 'Meta' || e.key === 'Control') {
      cmdHeld = false;
      if (hoveredA && origTitle) hoveredA.setAttribute('title', origTitle);
    }
  });
  // Reset cmdHeld when window loses focus (Cmd+Tab doesn't fire keyup)
  window.addEventListener('blur', function() {
    if (cmdHeld) {
      cmdHeld = false;
      if (hoveredA && origTitle) hoveredA.setAttribute('title', origTitle);
    }
  });

  document.addEventListener('keydown', function(e) {
    // Don't intercept keys when user is typing in an input field
    if (e.target && (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA')) return;
    var m = document.getElementById('hi-help-modal');
    if (e.key === 'Escape') {
      if (m && m.style.display !== 'none') { hiHelpToggle(false); return; }
    }
    if (m && m.style.display !== 'none') return;
    if (!activeEl) return;
    if (e.key === 'Enter') {
      e.preventDefault();
      window.location.href = activeEl.getAttribute('href');
      return;
    }
    if (e.key.startsWith('Arrow')) e.preventDefault();
    // Cmd+Down = back in history, Cmd+Up = forward in history
    if ((e.metaKey || e.ctrlKey) && e.key === 'ArrowDown' && backStack.length > 0) {
      activate(backStack.pop(), 'back'); flashActive();
      return;
    }
    if ((e.metaKey || e.ctrlKey) && e.key === 'ArrowUp' && fwdStack.length > 0) {
      activate(fwdStack.pop(), 'fwd'); flashActive();
      return;
    }
    var s = parseInt(activeEl.dataset.sosa);
    var handled = false;

    // Helper: get current dot position
    var dot = document.getElementById('s' + s);
    if (!dot) return;
    var cx = dot.offsetLeft + dot.offsetWidth / 2;
    var cy = dot.offsetTop + dot.offsetHeight / 2;

    if (e.shiftKey && e.key.startsWith('Arrow')) {
      // Shift+arrow: spatial navigation along connected lines, fallback to nearest
      var target = findConnected(s, cx, cy, e.key);
      if (target !== null) handled = activateSosa(target);
      if (!handled) { var nn = findNearest(cx, cy, e.key, s); if (nn) handled = activate(nn); }
    } else {
      var target = null;
      if (e.key === 'ArrowLeft') target = s * 2;
      else if (e.key === 'ArrowRight') target = s * 2 + 1;
      else if (e.key === 'ArrowDown') {
        var cpl = '%evarcpl;' === 'on';
        if (s > 1) {
          var child = Math.floor(s / 2);
          if (cpl && child % 2 === 1) child = child - 1;
          if (child >= (cpl ? 2 : 1)) {
            target = child;
          } else if (%dsosa; > 1) {
            sessionStorage.setItem('hi-active-sosa', String(%dsosa;));
            history.back();
            e.preventDefault(); return;
          }
        } else if (s === 1 && %dsosa; > 1) {
          sessionStorage.setItem('hi-active-sosa', String(%dsosa;));
          history.back();
          e.preventDefault(); return;
        }
      }
      else if (e.key === 'ArrowUp') {
        // Cycle through implex instances
        var pk2 = personKey(activeEl);
        var matches = Array.prototype.slice.call(tree.querySelectorAll('a[data-sosa]')).filter(function(a) {
          return personKey(a) === pk2;
        });
        if (matches.length > 1) {
          var idx = matches.indexOf(activeEl);
          var ni = (idx + 1 < matches.length) ? idx + 1 : 0;
          handled = activate(matches[ni]);
          if (handled) flashActive();
        }
        e.preventDefault(); return;
      }
      if (target !== null) handled = activateSosa(target);
      // No spatial fallback for genealogical nav: if father/mother/child doesn't exist, do nothing
      if (false && !handled && e.key.startsWith('Arrow')) {
        var nn = findNearest(cx, cy, e.key, s);
        if (nn) handled = activate(nn);
      }
    }
    if (handled) { flashActive(); e.preventDefault(); }

    // Home/End/PageUp/PageDown: default = vertical; Shift = horizontal
    var pageKeys = { Home:1, End:1, PageUp:1, PageDown:1 };
    if (pageKeys[e.key]) {
      if (e.shiftKey) {
        e.preventDefault();
        if (e.key === 'Home') window.scrollTo(0, window.scrollY);
        else if (e.key === 'End') window.scrollTo(document.documentElement.scrollWidth, window.scrollY);
        else if (e.key === 'PageUp') window.scrollBy(-window.innerWidth * 0.85, 0);
        else if (e.key === 'PageDown') window.scrollBy(window.innerWidth * 0.85, 0);
      }
    }
  });
})();
