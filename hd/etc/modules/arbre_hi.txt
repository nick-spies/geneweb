<script>
(function() {
  var tree = document.getElementById('hi-tree1');
  if (!tree) return;

  // Keyboard navigation
  // Keyboard navigation with class-based highlight (persists through button clicks)
  var activeEl = null;
  var infoEl = document.getElementById('hi-nav-info');

  var peerEls = [];
  var activeBox = null;
  var backStack = [];
  var fwdStack = [];
  // Person identity key: data-purl (Sosa box mode) or href (compact dot mode)
  function personKey(a) { return a.dataset.purl || a.getAttribute('href'); }
  // Get the highlight target: the parent box div in Sosa mode, the <a> itself in compact
  function hlTarget(a) {
    var box = document.getElementById('s' + a.dataset.sosa);
    return (box && box.querySelector('span')) ? box : a;
  }
  function activate(el, navDir, noScroll) {
    if (!el) return false;
    if (activeEl && activeEl !== el) {
      if (navDir === 'back') { fwdStack.push(activeEl); }
      else if (navDir === 'fwd') { backStack.push(activeEl); }
      else { backStack.push(activeEl); fwdStack = []; }
    }
    if (activeBox) { activeBox.classList.remove('hi-active'); activeBox.classList.remove('hi-glow'); }
    // clear previous peer highlights + glow
    for (var i = 0; i < peerEls.length; i++) { peerEls[i].classList.remove('hi-implex-peer'); peerEls[i].classList.remove('hi-glow'); }
    peerEls = [];
    var target = hlTarget(el);
    target.classList.add('hi-active');
    activeBox = target;
    activeEl = el;
    // highlight implex peers
    var pk = personKey(el);
    var peers = tree.querySelectorAll('a[data-sosa]');
    for (var i = 0; i < peers.length; i++) {
      if (peers[i] !== el && personKey(peers[i]) === pk) {
        var pt = hlTarget(peers[i]);
        pt.classList.add('hi-implex-peer');
        peerEls.push(pt);
      }
    }
    // persistent glow on active + peers
    target.classList.add('hi-glow');
    for (var i = 0; i < peerEls.length; i++) peerEls[i].classList.add('hi-glow');
    // scroll into view (skip for hover)
    if (!noScroll) {
      var scrollTarget = document.getElementById('s' + el.dataset.sosa) || target;
      scrollTarget.scrollIntoView({ behavior: 'smooth', block: 'center', inline: 'center' });
    }
    // update info bar
    if (infoEl) {
      var t = el.getAttribute('title') || '';
      infoEl.textContent = t.replace(/\n/g, ' \u00b7 ').replace(/( \u00b7 )+$/, '').replace(/^( \u00b7 )+/, '');
    }
    return true;
  }
  function activateSosa(n) {
    var el = tree.querySelector('a[data-sosa="' + n + '"]');
    return el ? activate(el) : false;
  }
  function flashNode(node) {
    node.classList.remove('hi-flash');
    void node.offsetWidth;
    node.classList.add('hi-flash');
    node.addEventListener('animationend', function() { node.classList.remove('hi-flash'); }, {once:true});
  }
  function flashActive() {
    if (activeBox) flashNode(activeBox);
    for (var i = 0; i < peerEls.length; i++) flashNode(peerEls[i]);
  }

  // Build node position cache for spatial navigation
  var nodeCache = [];
  function buildCache() {
    nodeCache = [];
    var els = tree.querySelectorAll('a[data-sosa]');
    for (var i = 0; i < els.length; i++) {
      var dot = document.getElementById('s' + els[i].dataset.sosa);
      if (!dot) continue;
      nodeCache.push({
        el: els[i],
        x: dot.offsetLeft + dot.offsetWidth / 2,
        y: dot.offsetTop + dot.offsetHeight / 2
      });
    }
  }

  // Get dot center position for a sosa number
  function dotPos(sosa) {
    var d = document.getElementById('s' + sosa);
    if (!d) return null;
    return { x: d.offsetLeft + d.offsetWidth / 2, y: d.offsetTop + d.offsetHeight / 2 };
  }

  // Spatial nav follows connected nodes: father(s*2), mother(s*2+1), child(s/2)
  // Maps arrow key to whichever connected node lies in that direction
  function findConnected(s, cx, cy, dir) {
    var neighbors = [];
    var f = dotPos(s * 2);
    if (f) neighbors.push({ sosa: s * 2, dx: f.x - cx, dy: f.y - cy });
    var m = dotPos(s * 2 + 1);
    if (m) neighbors.push({ sosa: s * 2 + 1, dx: m.x - cx, dy: m.y - cy });
    if (s > 1) {
      var c = dotPos(Math.floor(s / 2));
      if (c) neighbors.push({ sosa: Math.floor(s / 2), dx: c.x - cx, dy: c.y - cy });
    }
    // Score each neighbor: how well does it match the arrow direction?
    var best = null, bestScore = -Infinity;
    for (var i = 0; i < neighbors.length; i++) {
      var n = neighbors[i];
      var score = 0;
      if (dir === 'ArrowUp') score = -n.dy;
      else if (dir === 'ArrowDown') score = n.dy;
      else if (dir === 'ArrowLeft') score = -n.dx;
      else if (dir === 'ArrowRight') score = n.dx;
      if (score > 2 && score > bestScore) { bestScore = score; best = n.sosa; }
    }
    return best;
  }

  // Find nearest node in a screen direction from any position (fallback when no connected node)
  function findNearest(cx, cy, dir, currentSosa) {
    var best = null, bestDist = Infinity;
    for (var i = 0; i < nodeCache.length; i++) {
      var n = nodeCache[i];
      if (n.el.dataset.sosa === String(currentSosa)) continue;
      var dx = n.x - cx, dy = n.y - cy;
      var ok = false;
      if (dir === 'ArrowUp' && dy < -2) ok = true;
      else if (dir === 'ArrowDown' && dy > 2) ok = true;
      else if (dir === 'ArrowLeft' && dx < -2) ok = true;
      else if (dir === 'ArrowRight' && dx > 2) ok = true;
      if (!ok) continue;
      var along, cross;
      if (dir === 'ArrowUp' || dir === 'ArrowDown') {
        along = Math.abs(dy); cross = Math.abs(dx);
      } else {
        along = Math.abs(dx); cross = Math.abs(dy);
      }
      var dist = along + cross * 5;
      if (dist < bestDist) { bestDist = dist; best = n.el; }
    }
    return best;
  }

  // On load: mark all implex nodes blue and add cross-references to tooltips
  function markImplexes() {
    var byHref = {};
    var els = tree.querySelectorAll('a[data-sosa]');
    for (var i = 0; i < els.length; i++) {
      var h = personKey(els[i]);
      if (!byHref[h]) byHref[h] = [];
      byHref[h].push({ sosa: els[i].dataset.sosa, el: els[i] });
    }
    for (var h in byHref) {
      var group = byHref[h];
      if (group.length > 1) {
        var sosas = group.map(function(g) { return g.sosa; });
        for (var j = 0; j < group.length; j++) {
          // color dot blue
          var dot = document.getElementById('s' + group[j].sosa);
          if (dot) {
            dot.style.background = '#1e90ff';
            dot.style.color = '#ff0';
            var spans = dot.querySelectorAll('a, span');
            for (var k = 0; k < spans.length; k++) spans[k].style.color = '#ff0';
            dot.style.boxShadow = '0 0 3px 2px rgba(30,144,255,0.7)';
          }
          // add cross-ref to title if missing
          var el = group[j].el;
          var others = sosas.filter(function(s) { return s !== group[j].sosa; });
          var t = el.getAttribute('title') || '';
          var ref = ' (=' + others.join('|') + ')';
          // replace existing partial cross-ref or add new one
          t = t.replace(/ \(=[0-9,|]+\)/, '');
          // match sosa number with or without comma formatting
          var sn = group[j].sosa;
          var snc = Number(sn).toLocaleString();
          t = t.replace('Sosa ' + snc, 'Sosa ' + snc + ref);
          if (t.indexOf(ref) < 0) t = t.replace('Sosa ' + sn, 'Sosa ' + sn + ref);
          el.setAttribute('title', t);
        }
      }
    }
  }

  // Focus sosa 1 on load
  window.addEventListener('load', function() { activateSosa(1); buildCache(); markImplexes(); });

  // Sosa search: inject into top navbar + make navbar sticky
  (function() {
    var navBar = document.querySelector('nav.navbar');
    if (navBar) {
      navBar.style.position = 'fixed';
      navBar.style.top = '0';
      navBar.style.left = '0';
      navBar.style.right = '0';
      navBar.style.zIndex = '101';
      navBar.style.background = '#fff';
      // push content below fixed navbar
      document.body.style.paddingTop = navBar.offsetHeight + 'px';
      var tb = document.getElementById('hi-toolbar');
      if (tb) {
        tb.style.position = 'fixed';
        tb.style.top = navBar.offsetHeight + 'px';
        tb.style.left = '0';
        tb.style.right = '0';
        tb.style.zIndex = '100';
        document.body.style.paddingTop = (navBar.offsetHeight + tb.offsetHeight) + 'px';
      }
    }
    var nav = document.querySelector('nav.navbar ul.nav');
    if (!nav) return;
    var li = document.createElement('li');
    li.className = 'nav-item ml-2 d-flex align-items-center';
    li.style.cssText = 'white-space:nowrap;';
    li.innerHTML = '<i class="fa fa-search text-secondary mr-1"></i>' +
      '<input id="hi-sosa-search" type="text" inputmode="numeric" class="form-control form-control-sm d-inline-block" ' +
      'style="width:5em;height:1.8em;padding:1px 4px;font-size:13px;" title="Go to Sosa number">';
    nav.appendChild(li);
    document.getElementById('hi-sosa-search').addEventListener('keydown', function(e) {
      if (e.key !== 'Enter') return;
      e.preventDefault();
      var n = parseInt(this.value);
      if (!n || n < 1) return;
      if (activateSosa(n)) {
        flashActive();
      } else {
        this.style.background = '#fdd';
        var inp = this;
        setTimeout(function() { inp.style.background = ''; }, 800);
      }
    });
  })();

  // Hover dwell: hold pointer over a node to highlight it
  // If Cmd held on hover, show "Cmd+click: go to person" tooltip instead
  var hoverTimer = null;
  var hoveredA = null;
  var origTitle = '';
  var glowEls = [];
  function clearGlow() {
    for (var i = 0; i < glowEls.length; i++) glowEls[i].classList.remove('hi-glow');
    glowEls = [];
  }
  tree.addEventListener('mouseover', function(e) {
    var a = e.target.closest('a[data-sosa]');
    if (!a) return;
    clearTimeout(hoverTimer);
    hoverTimer = setTimeout(function() {
      if (activeEl === a) return;
      activate(a, null, true);
    }, 500);
    if (hoveredA !== a) {
      if (hoveredA) hoveredA.setAttribute('title', origTitle);
      hoveredA = a;
      origTitle = a.getAttribute('title') || '';
    }
    if (cmdHeld) {
      var name = origTitle.split('\n')[1] || '';
      a.setAttribute('title', '\u2318+click: go to ' + name.trim());
    } else if (hoveredA === a) {
      a.setAttribute('title', origTitle);
    }
  });
  tree.addEventListener('mouseout', function(e) {
    var a = e.target.closest('a[data-sosa]');
    if (!a) return;
    var related = e.relatedTarget;
    if (related && a.contains(related)) return;
    clearTimeout(hoverTimer);
    if (hoveredA === a) {
      a.setAttribute('title', origTitle);
      hoveredA = null; origTitle = '';
    }
  });

  // Cmd+click = go to person page; plain click = re-root chart (default href)
  tree.addEventListener('click', function(e) {
    var a = e.target.closest('a[data-sosa]');
    if (!a) return;
    if (e.metaKey || e.ctrlKey) {
      e.preventDefault();
      var url = a.dataset.purl;
      if (url) window.location.href = url;
    }
  });

  // Track Cmd key state for tooltip and info bar hints
  var cmdHeld = false;
  document.addEventListener('keydown', function(e) {
    if (e.key === 'Meta' || e.key === 'Control') cmdHeld = true;
  });
  document.addEventListener('keyup', function(e) {
    if (e.key === 'Meta' || e.key === 'Control') cmdHeld = false;
  });

  document.addEventListener('keydown', function(e) {
    if (!activeEl) return;
    if (e.key.startsWith('Arrow')) e.preventDefault();
    // Cmd+Down = back in history, Cmd+Up = forward in history
    if ((e.metaKey || e.ctrlKey) && e.key === 'ArrowDown' && backStack.length > 0) {
      activate(backStack.pop(), 'back'); flashActive();
      return;
    }
    if ((e.metaKey || e.ctrlKey) && e.key === 'ArrowUp' && fwdStack.length > 0) {
      activate(fwdStack.pop(), 'fwd'); flashActive();
      return;
    }
    var s = parseInt(activeEl.dataset.sosa);
    var handled = false;

    // Helper: get current dot position
    var dot = document.getElementById('s' + s);
    if (!dot) return;
    var cx = dot.offsetLeft + dot.offsetWidth / 2;
    var cy = dot.offsetTop + dot.offsetHeight / 2;

    if (e.shiftKey && e.key.startsWith('Arrow')) {
      // Shift+arrow: spatial navigation along connected lines, fallback to nearest
      var target = findConnected(s, cx, cy, e.key);
      if (target !== null) handled = activateSosa(target);
      if (!handled) { var nn = findNearest(cx, cy, e.key, s); if (nn) handled = activate(nn); }
    } else {
      var target = null;
      if (e.key === 'ArrowLeft') target = s * 2;
      else if (e.key === 'ArrowRight') target = s * 2 + 1;
      else if (e.key === 'ArrowDown' && s > 1) target = Math.floor(s / 2);
      else if (e.key === 'ArrowUp') {
        // Cycle through implex instances
        var pk2 = personKey(activeEl);
        var matches = Array.prototype.slice.call(tree.querySelectorAll('a[data-sosa]')).filter(function(a) {
          return personKey(a) === pk2;
        });
        if (matches.length > 1) {
          var idx = matches.indexOf(activeEl);
          var ni = (idx + 1 < matches.length) ? idx + 1 : 0;
          handled = activate(matches[ni]);
          if (handled) flashActive();
        }
        e.preventDefault(); return;
      }
      if (target !== null) handled = activateSosa(target);
      // No spatial fallback for genealogical nav: if father/mother/child doesn't exist, do nothing
      if (false && !handled && e.key.startsWith('Arrow')) {
        var nn = findNearest(cx, cy, e.key, s);
        if (nn) handled = activate(nn);
      }
    }
    if (handled) { flashActive(); e.preventDefault(); }
  });
})();
</script>
